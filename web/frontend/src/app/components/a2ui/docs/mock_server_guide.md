# A2UI Backend Mock Server Guide

This guide shows how to create a mock backend server that responds with A2UI payloads following the A2A protocol.

## Quick Start Mock Server (Node.js)

Create a simple Express server that streams A2UI responses:

```javascript
// mock-server.js
const express = require('express');
const cors = require('cors');

const app = express();
app.use(cors());
app.use(express.json());

app.post('/api/agent', async (req, res) => {
  // Set SSE headers
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  const { params } = req.body;
  const userMessage = params?.message?.parts?.[0]?.text || '';

  // Send acknowledgment
  const ackMessage = {
    kind: 'message',
    message: {
      role: 'agent',
      messageId: `msg-${Date.now()}`,
      parts: [
        {
          kind: 'text',
          text: `I received your message: "${userMessage}". Generating response...`
        }
      ]
    }
  };
  res.write(`data: ${JSON.stringify(ackMessage)}\n\n`);

  // Simulate processing delay
  await new Promise(resolve => setTimeout(resolve, 1000));

  // Send beginRendering
  const beginRendering = {
    kind: 'message',
    message: {
      role: 'agent',
      messageId: `msg-${Date.now()}`,
      parts: [
        {
          kind: 'data',
          data: {
            beginRendering: {
              surfaceId: 'response-surface',
              root: 'response-card'
            }
          }
        }
      ]
    }
  };
  res.write(`data: ${JSON.stringify(beginRendering)}\n\n`);

  await new Promise(resolve => setTimeout(resolve, 500));

  // Send surfaceUpdate with A2UI components
  const surfaceUpdate = {
    kind: 'message',
    message: {
      role: 'agent',
      messageId: `msg-${Date.now()}`,
      parts: [
        {
          kind: 'data',
          data: {
            surfaceUpdate: {
              surfaceId: 'response-surface',
              catalog: 'inline',
              components: [
                {
                  Card: {
                    id: 'response-card',
                    slots: {
                      children: [
                        {
                          Column: {
                            id: 'main-col',
                            slots: {
                              children: [
                                {
                                  Text: {
                                    id: 'title',
                                    text: 'Agent Response',
                                    usageHint: 'h2'
                                  }
                                },
                                {
                                  Divider: {
                                    id: 'div-1',
                                    axis: 'horizontal'
                                  }
                                },
                                {
                                  Text: {
                                    id: 'response-text',
                                    text: `You asked: "${userMessage}". This is a dynamic A2UI response generated by the agent.`,
                                    usageHint: 'body'
                                  }
                                },
                                {
                                  Button: {
                                    id: 'action-btn',
                                    primary: true,
                                    action: {
                                      name: 'confirm_action'
                                    },
                                    slots: {
                                      child: {
                                        Text: {
                                          id: 'btn-text',
                                          text: 'Acknowledge'
                                        }
                                      }
                                    }
                                  }
                                }
                              ]
                            }
                          }
                        }
                      ]
                    }
                  }
                }
              ]
            }
          }
        }
      ]
    }
  };
  res.write(`data: ${JSON.stringify(surfaceUpdate)}\n\n`);

  // Send completion
  res.write('data: [DONE]\n\n');
  res.end();
});

const PORT = 8080;
app.listen(PORT, () => {
  console.log(`Mock A2A server running on http://localhost:${PORT}`);
});
```

Install dependencies and run:

```bash
npm install express cors
node mock-server.js
```

## Python Mock Server (FastAPI)

```python
# mock_server.py
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import json
import asyncio
from typing import List, Dict, Any

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

class MessagePart(BaseModel):
    kind: str
    text: str = None
    data: Dict[str, Any] = None

class Message(BaseModel):
    role: str
    parts: List[MessagePart]
    messageId: str = None

class RequestParams(BaseModel):
    message: Message

class A2ARequest(BaseModel):
    jsonrpc: str
    method: str
    params: RequestParams
    id: str

async def generate_stream(user_message: str):
    # Acknowledgment
    yield f'data: {json.dumps({
        "kind": "message",
        "message": {
            "role": "agent",
            "messageId": f"msg-{id(user_message)}",
            "parts": [{
                "kind": "text",
                "text": f"Received: {user_message}"
            }]
        }
    })}\n\n'
    
    await asyncio.sleep(1)
    
    # BeginRendering
    yield f'data: {json.dumps({
        "kind": "message",
        "message": {
            "role": "agent",
            "parts": [{
                "kind": "data",
                "data": {
                    "beginRendering": {
                        "surfaceId": "response-surface",
                        "root": "response-card"
                    }
                }
            }]
        }
    })}\n\n'
    
    await asyncio.sleep(0.5)
    
    # SurfaceUpdate
    yield f'data: {json.dumps({
        "kind": "message",
        "message": {
            "role": "agent",
            "parts": [{
                "kind": "data",
                "data": {
                    "surfaceUpdate": {
                        "surfaceId": "response-surface",
                        "catalog": "inline",
                        "components": [{
                            "Card": {
                                "id": "response-card",
                                "slots": {
                                    "children": [{
                                        "Column": {
                                            "id": "col-1",
                                            "slots": {
                                                "children": [
                                                    {
                                                        "Text": {
                                                            "id": "title",
                                                            "text": "City Simulator Response",
                                                            "usageHint": "h2"
                                                        }
                                                    },
                                                    {
                                                        "Text": {
                                                            "id": "body",
                                                            "text": f"Processing: {user_message}",
                                                            "usageHint": "body"
                                                        }
                                                    }
                                                ]
                                            }
                                        }
                                    }]
                                }
                            }
                        }]
                    }
                }
            }]
        }
    })}\n\n'
    
    yield 'data: [DONE]\n\n'

@app.post("/api/agent")
async def agent_endpoint(request: A2ARequest):
    user_text = request.params.message.parts[0].text if request.params.message.parts else ""
    return StreamingResponse(
        generate_stream(user_text),
        media_type="text/event-stream"
    )

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8080)
```

Run:

```bash
pip install fastapi uvicorn
python mock_server.py
```


4. Send a message in the chat interface
5. Verify you receive the A2UI surface response

## Response Format Checklist

✓ Use `text/event-stream` content type  
✓ Each message prefixed with `data: `  
✓ Each message ends with `\n\n`  
✓ Messages contain `kind: "message"`  
✓ Parts array contains `kind: "text"` or `kind: "data"`  
✓ A2UI payloads wrapped in `data` field  
✓ Always send `beginRendering` before `surfaceUpdate`  
✓ Component IDs are unique and referenced correctly  

## Advanced: Multi-Agent Orchestration

Show tool call pills during processing:

```javascript
// Send tool call indicator
const toolCall = {
  kind: 'message',
  message: {
    role: 'agent',
    messageId: 'msg-123',
    parts: [{
      kind: 'data',
      data: {
        call: {
          agent: 'TrafficAnalyzer',
          status: 'calling'
        }
      }
    }]
  }
};
res.write(`data: ${JSON.stringify(toolCall)}\n\n`);

// Process...

// Send response indicator
const toolResponse = {
  kind: 'message',
  message: {
    role: 'agent',
    messageId: 'msg-123', // Same ID for update
    parts: [{
      kind: 'data',
      data: {
        response: {
          agent: 'TrafficAnalyzer',
          status: 'complete',
          result: 'Analysis complete'
        }
      }
    }]
  }
};
res.write(`data: ${JSON.stringify(toolResponse)}\n\n`);
```
